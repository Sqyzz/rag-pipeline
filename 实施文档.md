# 📌 实施文档

## 0. 项目元信息与约定

**项目名称**：enterprise-graphrag

**目标**：实现以下 pipeline：

1. VectorRAG baseline
2. Text MapReduce baseline
3. GraphRAG indexing + summaries
4. Query engine + evaluation framework

**核心需求**：

- 支持本地模型（Remote 3090 server）与在线 API 任意切换
- 基于 Mac 编写代码，通过统一接口调用远端推理
- 优先使用公开数据（如 Enron）作为主实验数据

**核心依赖**：

- LlamaIndex（KnowledgeGraphIndex）用于实体/关系抽取与基础 RAG
- FAISS/ChromaDB 作为向量检索库
- 自制 KG 结构用于实验消融
- Baseline 和消融代码必须可自动化执行

## 3. LLM 接口统一代码（`src/utils/llm_wrapper.py`）

给自动化助手直接写以下内容：

```
import requests, os
from .config import cfg

def llm_chat(messages):
    backend = cfg.llm.backend
    if backend == "local":
        base_url = cfg.llm.local.base_url
        model = cfg.llm.local.model
        key = None
    else:
        base_url = cfg.llm.api.base_url
        model = cfg.llm.api.model
        key = os.getenv(cfg.llm.api.api_key_env)

    url = f"{base_url}/chat/completions"
    headers = {}
    if key:
        headers["Authorization"] = f"Bearer {key}"

    payload = {"model": model, "messages": messages, "temperature": 0.2}
    r = requests.post(url, json=payload, headers=headers)
    r.raise_for_status()
    return r.json()["choices"][0]["message"]["content"]
```

## 4. Embedding 接口（`src/utils/embedder.py`）

写成统调用，方便后续评价和检索：

```
import numpy as np
from sentence_transformers import SentenceTransformer
from .config import cfg

_local_model = None

def embed_texts(texts):
    if cfg.embedding.backend == "local":
        global _local_model
        if _local_model is None:
            _local_model = SentenceTransformer(cfg.embedding.local.model)
        return _local_model.encode(texts)
    else:
        import requests, os
        url = f"{cfg.embedding.api.base_url}/embeddings"
        headers = {"Authorization": f"Bearer {os.getenv(cfg.embedding.api.api_key_env)}"}
        payload = {"model": cfg.embedding.api.model, "input": texts}
        r = requests.post(url, json=payload, headers=headers)
        r.raise_for_status()
        embs = r.json()["data"]
        return np.array([e["embedding"] for e in embs])
```

## 5. 数据加载 & Enron 处理（`src/ingestion/load_docs.py`）

```
import os, json
def load_enron(raw_dir, out_file):
    docs = []
    for root, _, files in os.walk(raw_dir):
        for f in files:
            if f.endswith(".txt"):
                with open(os.path.join(root, f),"r",encoding="utf8") as r:
                    text = r.read()
                docs.append({"doc_id":f, "text":text})
    with open(out_file,"w") as w:
        for doc in docs:
            w.write(json.dumps(doc,ensure_ascii=False)+"\n")
```

## 6. 文本 Chunking（`src/ingestion/chunking.py`）

```
import json, uuid

def chunk_texts(in_file, out_file, chunk_size, overlap):
    chunks = []
    with open(in_file) as r:
        for line in r:
            d=json.loads(line)
            text = d["text"]
            pos=0
            while pos < len(text):
                chunk = text[pos: pos+chunk_size]
                chunks.append({"chunk_id":str(uuid.uuid4()),"text":chunk})
                pos += chunk_size - overlap
    with open(out_file,"w") as w:
        for c in chunks:
            w.write(json.dumps(c,ensure_ascii=False)+"\n")
```

## 7. VectorRAG Baseline（`src/baselines/vector_rag.py`）

```
import faiss, json
from utils.embedder import embed_texts

def build_index(chunks_file, idx_file, store_file):
    texts=[]
    meta=[]
    with open(chunks_file) as r:
        for line in r:
            obj=json.loads(line)
            texts.append(obj["text"])
            meta.append(obj)
    embs=embed_texts(texts)
    dim=embs.shape[1]
    index=faiss.IndexFlatIP(dim)
    index.add(embs)
    faiss.write_index(index, idx_file)
    with open(store_file,"w") as w:
        json.dump(meta,w)

def retrieve_and_answer(query, idx_file, store_file, top_k):
    index=faiss.read_index(idx_file)
    with open(store_file) as r:
        store=json.load(r)
    q_emb=embed_texts([query])
    D,I=index.search(q_emb, top_k)
    contexts=[store[i]["text"] for i in I[0]]
    return contexts
```

## 8. Text MapReduce Baseline（`src/baselines/text_mapreduce.py`）

```
from utils.llm_wrapper import llm_chat
import json

def mapreduce_answer(query, chunks_file, map_count=10):
    # map
    with open(chunks_file) as r:
        all_chunks=[json.loads(l) for l in r]
    chunk_splits=[all_chunks[i::map_count] for i in range(map_count)]
    partial=[]
    for sp in chunk_splits:
        msgs=[{"role":"system","content":f"Query: {query}\nUse these contexts:{[c['text'] for c in sp]}"}]
        partial.append(llm_chat(msgs))
    # reduce
    msgs=[{"role":"system","content":f"Combine these partial answers:{partial}"}]
    return llm_chat(msgs)
```

## 9. 评测脚本（`src/evaluation/run_eval.py`）

```
import json
def simple_eval(pred_file, gold_file, out_csv):
    from difflib import SequenceMatcher
    res=[]
    with open(pred_file) as p, open(gold_file) as g:
        preds=list(p); golds=list(g)
    for pr,gl in zip(preds,golds):
        pr=json.loads(pr); gl=json.loads(gl)
        score=SequenceMatcher(None, pr["answer"],gl["answer"]).ratio()
        res.append({"qid":pr["qid"],"score":score})
    import pandas as pd
    pd.DataFrame(res).to_csv(out_csv,index=False)
```
